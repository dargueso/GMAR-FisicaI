# Coriolis 3D Simulator Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a self-contained 3D Coriolis simulator with a textured spherical Earth, 4 reference frame modes, drag-to-orbit, and click-to-place, matching the existing simulator's dark design.

**Architecture:** Three.js r158 (CDN) renders one scene; physics integrates in Earth-fixed coords; four camera modes transform Earth group rotation and camera position each rAF frame; orbit is custom theta/phi (no OrbitControls dependency needed); trail stores both inertial and Earth-fixed arrays, switching on frame mode.

**Tech Stack:** HTML5, vanilla JavaScript (ES6), Three.js r158 (CDN), Node.js assert for physics unit tests.

---

## Coordinate System (read before every task)

- Three.js Y-up, right-handed. Earth centre = origin.
- North Pole = `(0, 1, 0)`. Prime meridian equator = `(1, 0, 0)`.
- lat/lon → unit vec: `x = cos(lat)·cos(lon)`, `y = sin(lat)`, `z = -cos(lat)·sin(lon)`
- East = `-z` direction at prime meridian equator.
- All physics positions are **unit vectors** (sphere radius = 1). Velocities are **m/s** (scaled by `EARTH_RADIUS` when needed).
- Earth rotation angle: `θ = 2π × elapsedReal / (periodH × 3600)` rad, always positive (counterclockwise NH = increasing θ).

## Display Frame Rules

| `frameMode` | `earthGroup.rotation.y` | Particle / trail world coords |
|---|---|---|
| `'fixed'` | `θ` | inertial = `rotateY(earthFixed, θ)` |
| `'earth'` | `0` | Earth-fixed directly |
| `'follow'` | `θ` | inertial = `rotateY(earthFixed, θ)` |
| `'followrot'` | `0` | Earth-fixed directly |

```js
// simulations/coriolis3d/physics3d.js

const EARTH_RADIUS = 6_371_000; // metres

// --- Vector math (plain objects for Node.js testability) ---

function add3(a, b)    { return { x: a.x+b.x, y: a.y+b.y, z: a.z+b.z }; }
function sub3(a, b)    { return { x: a.x-b.x, y: a.y-b.y, z: a.z-b.z }; }
function scale3(v, s)  { return { x: v.x*s,   y: v.y*s,   z: v.z*s   }; }
function dot3(a, b)    { return a.x*b.x + a.y*b.y + a.z*b.z; }
function mag3(v)       { return Math.sqrt(dot3(v, v)); }
function norm3(v)      { return scale3(v, 1 / mag3(v)); }

function cross3(a, b) {
  return {
    x: a.y*b.z - a.z*b.y,
    y: a.z*b.x - a.x*b.z,
    z: a.x*b.y - a.y*b.x,
  };
}

// Remove radial component of v (project onto tangent plane at unit-sphere pos p)
function projectTangent(v, p) {
  const radial = dot3(v, p);
  return sub3(v, scale3(p, radial));
}

// Rotate vector around Y axis by angle theta (counterclockwise = NH Earth rotation)
function rotateY(v, theta) {
  const c = Math.cos(theta), s = Math.sin(theta);
  return { x: v.x*c + v.z*s, y: v.y, z: -v.x*s + v.z*c };
}

// lat/lon (degrees) → unit vector
function latLonToVec3(lat, lon) {
  const φ = lat * Math.PI / 180;
  const λ = lon * Math.PI / 180;
  return { x: Math.cos(φ)*Math.cos(λ), y: Math.sin(φ), z: -Math.cos(φ)*Math.sin(λ) };
}

// Unit vector → { lat, lon } in degrees
function vec3ToLatLon(v) {
  const r   = mag3(v);
  const lat = Math.asin(v.y / r) * 180 / Math.PI;
  const lon = Math.atan2(-v.z, v.x) * 180 / Math.PI;
  return { lat, lon };
}

// Convert compass bearing (deg, 0=N, 90=E) + speed (m/s) to 3D velocity at surface pos
function bearingSpeedToVel3D(pos, bearing, speed) {
  const up      = { x: 0, y: 1, z: 0 };
  // North tangent: project (0,1,0) onto tangent plane, normalise
  const northT  = norm3(projectTangent(up, pos));
  // East tangent: pos × northT (right-hand rule gives westward, negate for East)
  const eastT   = scale3(cross3(pos, northT), -1);
  const rad     = bearing * Math.PI / 180;
  return add3(scale3(eastT, speed * Math.sin(rad)), scale3(northT, speed * Math.cos(rad)));
}

// RK4 integration step on sphere surface.
// pos: unit vector (Earth-fixed)
// vel: m/s 3D velocity (tangent to sphere, Earth-fixed frame)
// omegaVec: {x,y,z} rad/s  (Earth rotation axis, e.g. {x:0, y:Ωz, z:0})
// dt: real seconds
// Returns: { pos, vel } — pos normalised to unit sphere, vel projected to tangent plane
function rk4Step3D(pos, vel, omegaVec, dt) {
  // Work in unit-sphere velocity units: vU = vel / EARTH_RADIUS (unit/s)
  // Coriolis: a = -2 * omega × vU  (in unit/s²)
  // The factor EARTH_RADIUS cancels: omega × (vel/R) has units rad/s × 1/s = 1/s²
  // This equals (omega × vel) / R = (m/s²) / R — the correct unit-sphere acceleration.

  const vU0 = scale3(vel, 1 / EARTH_RADIUS);

  function deriv(p, v) {
    const cor   = cross3(omegaVec, v);           // omega × v (unit/s²)
    const accel = scale3(cor, -2);               // -2 omega × v
    return { dp: v, dv: projectTangent(accel, p) };
  }

  const k1 = deriv(pos, vU0);
  const p2 = norm3(add3(pos, scale3(k1.dp, dt/2)));
  const v2 = add3(vU0, scale3(k1.dv, dt/2));
  const k2 = deriv(p2, v2);
  const p3 = norm3(add3(pos, scale3(k2.dp, dt/2)));
  const v3 = add3(vU0, scale3(k2.dv, dt/2));
  const k3 = deriv(p3, v3);
  const p4 = norm3(add3(pos, scale3(k3.dp, dt)));
  const v4 = add3(vU0, scale3(k3.dv, dt));
  const k4 = deriv(p4, v4);

  const newPosRaw = add3(pos, scale3(
    add3(add3(k1.dp, scale3(k2.dp, 2)), add3(scale3(k3.dp, 2), k4.dp)), dt/6));
  const newVU = add3(vU0, scale3(
    add3(add3(k1.dv, scale3(k2.dv, 2)), add3(scale3(k3.dv, 2), k4.dv)), dt/6));

  const newPos = norm3(newPosRaw);
  const newVel = scale3(projectTangent(newVU, newPos), EARTH_RADIUS);
  return { pos: newPos, vel: newVel };
}

if (typeof module !== 'undefined') {
  module.exports = {
    EARTH_RADIUS,
    add3, sub3, scale3, dot3, mag3, norm3, cross3,
    projectTangent, rotateY,
    latLonToVec3, vec3ToLatLon,
    bearingSpeedToVel3D,
    rk4Step3D,
  };
}
```

```js
// simulations/coriolis3d/scene3d.js

// ── Constants ────────────────────────────────────────────────────────────────

const SURFACE_RADIUS = 1.005; // particle sits just above surface
const TRAIL_MAX      = 8000;  // max trail points

// ── Helpers ──────────────────────────────────────────────────────────────────

// Convert Earth-fixed plain {x,y,z} to THREE.Vector3
function toV3(p) { return new THREE.Vector3(p.x, p.y, p.z); }

// Apply Earth rotation (for Fixed/Follow modes)
function earthToWorld(p, earthAngle) {
  return toV3(p).applyEuler(new THREE.Euler(0, earthAngle, 0));
}

// ── Scene initialisation ─────────────────────────────────────────────────────

function initScene(canvas) {
  // Renderer
  const renderer = new THREE.WebGLRenderer({ canvas, antialias: true, alpha: false });
  renderer.setPixelRatio(Math.min(window.devicePixelRatio, 2));
  renderer.setSize(canvas.clientWidth, canvas.clientHeight, false);
  renderer.setClearColor(0x0f1117);

  // Scene
  const scene = new THREE.Scene();

  // Camera
  const camera = new THREE.PerspectiveCamera(45, 1, 0.01, 100);
  camera.position.set(0, 0.4, 3);
  camera.lookAt(0, 0, 0);

  // ── Earth group (rotated in Fixed/Follow modes) ───────────────────────────
  const earthGroup = new THREE.Group();
  scene.add(earthGroup);

  // Earth sphere
  const earthGeo = new THREE.SphereGeometry(1, 64, 64);
  const loader   = new THREE.TextureLoader();
  const earthMat = new THREE.MeshPhongMaterial({
    color:    0xffffff,
    specular: new THREE.Color(0x111111),
    shininess: 8,
  });
  loader.load(
    'https://unpkg.com/three@0.158.0/examples/textures/planets/earth_atmos_2048.jpg',
    (tex) => { earthMat.map = tex; earthMat.needsUpdate = true; },
    undefined,
    () => { earthMat.color.set(0x1a3a4a); } // offline fallback colour
  );
  const earthMesh = new THREE.Mesh(earthGeo, earthMat);
  earthGroup.add(earthMesh);

  // Atmosphere glow
  const atmGeo = new THREE.SphereGeometry(1.025, 32, 32);
  const atmMat = new THREE.MeshBasicMaterial({
    color: 0x1a6699,
    transparent: true,
    opacity: 0.15,
    side: THREE.BackSide,
  });
  earthGroup.add(new THREE.Mesh(atmGeo, atmMat));

  // Graticule (lat/lon grid)
  earthGroup.add(_buildGraticule());

  // ── Lighting (in scene = inertial space, sun stays fixed) ────────────────
  const sunLight = new THREE.DirectionalLight(0xffffff, 1.2);
  sunLight.position.set(5, 3, 5);
  scene.add(sunLight);
  scene.add(new THREE.AmbientLight(0xffffff, 0.15));

  // Resize handler
  function onResize() {
    const w = canvas.clientWidth;
    const h = canvas.clientHeight;
    renderer.setSize(w, h, false);
    camera.aspect = w / h;
    camera.updateProjectionMatrix();
  }
  new ResizeObserver(onResize).observe(canvas.parentElement);

  return { renderer, scene, camera, earthGroup, earthMesh };
}

// ── Graticule builder ─────────────────────────────────────────────────────────

function _buildGraticule() {
  const segs = [];
  const N = 90; // segments per circle/line

  function pushSeg(ax, ay, az, bx, by, bz) {
    segs.push(ax, ay, az, bx, by, bz);
  }

  // Latitude circles every 30° (skip poles)
  for (let latDeg = -60; latDeg <= 60; latDeg += 30) {
    const φ = latDeg * Math.PI / 180;
    for (let i = 0; i < N; i++) {
      const λ0 = (i / N) * 2 * Math.PI;
      const λ1 = ((i + 1) / N) * 2 * Math.PI;
      pushSeg(
        Math.cos(φ)*Math.cos(λ0), Math.sin(φ), -Math.cos(φ)*Math.sin(λ0),
        Math.cos(φ)*Math.cos(λ1), Math.sin(φ), -Math.cos(φ)*Math.sin(λ1)
      );
    }
  }

  // Longitude lines every 30°
  for (let lonDeg = 0; lonDeg < 360; lonDeg += 30) {
    const λ = lonDeg * Math.PI / 180;
    for (let i = 0; i < N; i++) {
      const φ0 = (-90 + (i / N) * 180) * Math.PI / 180;
      const φ1 = (-90 + ((i + 1) / N) * 180) * Math.PI / 180;
      pushSeg(
```

```js
// ── Per-frame update functions ────────────────────────────────────────────────

// Update earthGroup rotation based on frame mode.
function updateEarthRotation(earthGroup, frameMode, earthAngle) {
  earthGroup.rotation.y = (frameMode === 'fixed' || frameMode === 'follow') ? earthAngle : 0;
}

// Compute world position of an Earth-fixed point for the current frame mode.
function getWorldPos(earthFixedPt, frameMode, earthAngle) {
  if (frameMode === 'fixed' || frameMode === 'follow') {
    return earthToWorld(earthFixedPt, earthAngle);
  }
  return toV3(earthFixedPt);
}

// Update particle mesh and glow position.
function updateParticle(partMesh, partGlow, earthFixedPos, frameMode, earthAngle) {
  const wp = getWorldPos(earthFixedPos, frameMode, earthAngle).multiplyScalar(SURFACE_RADIUS);
  partMesh.position.copy(wp);
  partGlow.position.copy(wp);
}

// Update crosshair position. Sits on surface at initPos.
function updateCrosshair(crosshairLine, initPos, frameMode, earthAngle) {
  const wp = getWorldPos(initPos, frameMode, earthAngle).normalize();
  crosshairLine.position.copy(wp);
  // Orient crosshair tangent to sphere surface (lookAt from centre)
  crosshairLine.quaternion.setFromUnitVectors(new THREE.Vector3(0, 1, 0), wp.clone());
}

// Update trail line geometry from stored arrays.
// trailEF: Earth-fixed array of {x,y,z}
// trailIn: inertial array of {x,y,z}  (computed at step time)
function updateTrail(trailLine, trailBuf, trailEF, trailIn, frameMode) {
  const trail = (frameMode === 'fixed' || frameMode === 'follow') ? trailIn : trailEF;
  const count = Math.min(trail.length, TRAIL_MAX);
  for (let i = 0; i < count; i++) {
    trailBuf[i*3]   = trail[i].x;
    trailBuf[i*3+1] = trail[i].y;
    trailBuf[i*3+2] = trail[i].z;
  }
  trailLine.geometry.attributes.position.needsUpdate = true;
  trailLine.geometry.setDrawRange(0, count);
}

// Update velocity and Coriolis arrows.
// earthFixedVel: {x,y,z} m/s in Earth-fixed frame
// omegaVec: {x,y,z} rad/s
function updateArrows(velArrow, corArrow, earthFixedPos, earthFixedVel, frameMode, earthAngle, omegaVec) {
  const wp  = getWorldPos(earthFixedPos, frameMode, earthAngle).multiplyScalar(SURFACE_RADIUS);
  const speed = Math.sqrt(earthFixedVel.x**2 + earthFixedVel.y**2 + earthFixedVel.z**2);

  // Velocity arrow: direction = vel in display frame, length proportional to speed (clamped)
  const velWorld = (frameMode === 'fixed' || frameMode === 'follow')
    ? earthToWorld(earthFixedVel, earthAngle)
    : toV3(earthFixedVel);
  if (speed > 0.1) {
    const velLen = Math.max(0.05, Math.min(0.3, speed / 2000));
    velArrow.position.copy(wp);
    velArrow.setDirection(velWorld.normalize());
    velArrow.setLength(velLen, velLen * 0.4, velLen * 0.25);
    velArrow.visible = true;
  } else {
    velArrow.visible = false;
  }

  // Coriolis arrow: only in rotating-frame modes (Earth, Follow+Rot)
  const showCor = (frameMode === 'earth' || frameMode === 'followrot');
  if (showCor && speed > 0.1) {
    // a_coriolis = -2 * omega × vel (in Earth-fixed frame)
    const omega2 = { x: omegaVec.x*2, y: omegaVec.y*2, z: omegaVec.z*2 };
    const cor = {
      x: -(omega2.y * earthFixedVel.z - omega2.z * earthFixedVel.y),
      y: -(omega2.z * earthFixedVel.x - omega2.x * earthFixedVel.z),
      z: -(omega2.x * earthFixedVel.y - omega2.y * earthFixedVel.x),
    };
    const corMag = Math.sqrt(cor.x**2 + cor.y**2 + cor.z**2);
    if (corMag > 1e-6) {
      const corLen = Math.max(0.05, Math.min(0.3, corMag / 0.05));
      corArrow.position.copy(wp);
      corArrow.setDirection(new THREE.Vector3(cor.x, cor.y, cor.z).normalize());
      corArrow.setLength(corLen, corLen * 0.4, corLen * 0.25);
      corArrow.visible = true;
    } else { corArrow.visible = false; }
  } else {
    corArrow.visible = false;
  }
}
```

```js
// Orbit state: theta (horizontal), phi (vertical), radius (zoom)
function initOrbit(canvas) {
  const orbit = {
    theta:      0.3,
    phi:        0.25,
    radius:     3.0,
    isDragging: false,
    didDrag:    false,
    lastX:      0,
    lastY:      0,
  };

  canvas.addEventListener('mousedown', (e) => {
    orbit.isDragging = true;
    orbit.didDrag    = false;
    orbit.lastX = e.clientX;
    orbit.lastY = e.clientY;
  });

  canvas.addEventListener('mousemove', (e) => {
    if (!orbit.isDragging) return;
    const dx = e.clientX - orbit.lastX;
    const dy = e.clientY - orbit.lastY;
    if (Math.abs(dx) > 2 || Math.abs(dy) > 2) orbit.didDrag = true;
    orbit.theta -= dx * 0.006;
    orbit.phi   -= dy * 0.006;
    orbit.phi    = Math.max(-Math.PI/2 + 0.05, Math.min(Math.PI/2 - 0.05, orbit.phi));
    orbit.lastX  = e.clientX;
    orbit.lastY  = e.clientY;
  });

  canvas.addEventListener('mouseup',    () => { orbit.isDragging = false; });
  canvas.addEventListener('mouseleave', () => { orbit.isDragging = false; });

  canvas.addEventListener('wheel', (e) => {
    orbit.radius = Math.max(1.5, Math.min(10, orbit.radius + e.deltaY * 0.004));
  }, { passive: true });

  return orbit;
}

// Reset orbit to default view (called on Reset button)
function resetOrbit(orbit) {
  orbit.theta  = 0.3;
  orbit.phi    = 0.25;
  orbit.radius = 3.0;
}

// Recompute camera position each frame based on reference frame mode.
// earthFixedPos: plain {x,y,z} unit vector (particle position on sphere)
// earthAngle: current Earth rotation angle (radians)
function updateCamera(camera, frameMode, earthFixedPos, earthAngle, orbit) {
  // Compute orbit target
  let target = new THREE.Vector3(0, 0, 0);

  if (frameMode === 'follow') {
    // Track particle in inertial space
    target.set(earthFixedPos.x, earthFixedPos.y, earthFixedPos.z)
          .applyEuler(new THREE.Euler(0, earthAngle, 0));
  } else if (frameMode === 'followrot') {
    // Track particle in Earth-fixed space
    target.set(earthFixedPos.x, earthFixedPos.y, earthFixedPos.z);
  }
  // 'fixed' and 'earth' modes: target = origin (0,0,0)

  // Camera position = target + spherical offset from orbit angles
  const phi_polar = Math.PI/2 - orbit.phi; // polar angle from +Y
  const camOffset = new THREE.Vector3(
    orbit.radius * Math.sin(phi_polar) * Math.sin(orbit.theta),
    orbit.radius * Math.cos(phi_polar),
    orbit.radius * Math.sin(phi_polar) * Math.cos(orbit.theta)
  );

  camera.position.copy(target).add(camOffset);
  camera.lookAt(target);
  camera.up.set(0, 1, 0);
}
```

```js
// simulations/coriolis3d/main3d.js

const TIME_SCALE_3D = 3600; // 1 animation second = 1 real hour

// Default start position: 45°N, lon=0°
const DEFAULT_LAT = 45, DEFAULT_LON = 0;

const state3D = {
  // Controls
  periodH:    23.93,   // Earth rotation period in hours
  speed:      200,     // m/s
  bearing:    90,      // degrees
  frameMode:  'fixed', // 'fixed' | 'earth' | 'follow' | 'followrot'

  // Physics
  pos:        null,    // {x,y,z} unit vector — Earth-fixed
  vel:        null,    // {x,y,z} m/s — Earth-fixed
  initPos:    null,    // {x,y,z} unit vector — starting geographic position
  elapsedReal: 0,      // accumulated real seconds

  // Trail storage (both frames for mode-switching)
  trailEF:    [],      // Earth-fixed points {x,y,z}
  trailIn:    [],      // inertial points {x,y,z} (computed at step time)

  running:    false,
  rafId:      null,
  lastTime:   null,
};

// Three.js objects (set by initAll3D)
let g_renderer, g_scene, g_camera, g_earthGroup, g_earthMesh;
let g_partMesh, g_partGlow, g_trailLine, g_trailBuf;
let g_velArrow, g_corArrow, g_crosshairLine, g_orbit;

function getOmegaVec() {
  const Ωz = 2 * Math.PI / (state3D.periodH * 3600);
  return { x: 0, y: Ωz, z: 0 };
}

function earthAngle() {
  return (2 * Math.PI * state3D.elapsedReal) / (state3D.periodH * 3600);
}

function resetSim3D() {
  const pos = state3D.initPos || latLonToVec3(DEFAULT_LAT, DEFAULT_LON);
  const vel = bearingSpeedToVel3D(pos, state3D.bearing, state3D.speed);
  state3D.pos         = { ...pos };
  state3D.vel         = { ...vel };
  state3D.initPos     = { ...pos };
  state3D.trailEF     = [];
  state3D.trailIn     = [];
  state3D.elapsedReal = 0;
  state3D.lastTime    = null;
  renderFrame3D();
}

function renderFrame3D() {
  const θ = earthAngle();

  updateEarthRotation(g_earthGroup, state3D.frameMode, θ);
  updateParticle(g_partMesh, g_partGlow, state3D.pos, state3D.frameMode, θ);
  updateCrosshair(g_crosshairLine, state3D.initPos, state3D.frameMode, θ);
  updateTrail(g_trailLine, g_trailBuf, state3D.trailEF, state3D.trailIn, state3D.frameMode);
  updateArrows(g_velArrow, g_corArrow, state3D.pos, state3D.vel, state3D.frameMode, θ, getOmegaVec());
  updateCamera(g_camera, state3D.frameMode, state3D.pos, θ, g_orbit);
  g_renderer.render(g_scene, g_camera);
  updateInfoBox();
}

function updateInfoBox() {
  const { lat, lon } = vec3ToLatLon(state3D.pos);
  const latStr = `${Math.abs(lat).toFixed(1)}°${lat >= 0 ? 'N' : 'S'}`;
  const lonStr = `${Math.abs(lon).toFixed(1)}°${lon >= 0 ? 'E' : 'W'}`;
  document.getElementById('info-latlon').textContent  = `Lat: ${latStr}  Lon: ${lonStr}`;
  document.getElementById('info-time').textContent    = `Time:   ${(state3D.elapsedReal / 3600).toFixed(2)} h`;
  document.getElementById('info-period').textContent  = `Period: ${state3D.periodH.toFixed(2)} h`;
  const spd = Math.sqrt(state3D.vel.x**2 + state3D.vel.y**2 + state3D.vel.z**2);
  document.getElementById('info-speed').textContent   = `Speed:  ${spd.toFixed(0)} m/s`;
}

function animate3D(timestamp) {
  if (!state3D.running) return;
  if (state3D.lastTime === null) state3D.lastTime = timestamp;

  const dtAnim = Math.min((timestamp - state3D.lastTime) / 1000, 0.05);
  state3D.lastTime = timestamp;
  const dtReal = dtAnim * TIME_SCALE_3D;

  // Physics step in Earth-fixed rotating frame
  const { pos: newPos, vel: newVel } = rk4Step3D(
    state3D.pos, state3D.vel, getOmegaVec(), dtReal
  );
  state3D.pos = newPos;
  state3D.vel = newVel;
  state3D.elapsedReal += dtReal;

  // Append trail points
  if (state3D.trailEF.length < TRAIL_MAX) {
    const θ = earthAngle();
    // Earth-fixed point
    state3D.trailEF.push({ x: newPos.x, y: newPos.y, z: newPos.z });
    // Inertial point: rotate Earth-fixed by current θ
    const inertial = rotateY(newPos, θ);
    state3D.trailIn.push({ x: inertial.x, y: inertial.y, z: inertial.z });
  }

  renderFrame3D();
  state3D.rafId = requestAnimationFrame(animate3D);
}

function startSim3D() {
  if (state3D.running) return;
  if (!state3D.pos) resetSim3D();
  state3D.running  = true;
  state3D.lastTime = null;
  state3D.rafId    = requestAnimationFrame(animate3D);
}

function pauseSim3D() {
  state3D.running = false;
  if (state3D.rafId) cancelAnimationFrame(state3D.rafId);
}

function initAll3D() {
  const canvas = document.getElementById('canvas3d');
  const sceneData = initScene(canvas);
  g_renderer  = sceneData.renderer;
  g_scene     = sceneData.scene;
  g_camera    = sceneData.camera;
  g_earthGroup = sceneData.earthGroup;
  g_earthMesh  = sceneData.earthMesh;

  const { partMesh, partGlow } = buildParticle(g_scene);
  g_partMesh = partMesh;
  g_partGlow = partGlow;

  const { trailLine, trailBuf } = buildTrail(g_scene);
  g_trailLine = trailLine;
  g_trailBuf  = trailBuf;

  const { velArrow, corArrow } = buildArrows(g_scene);
  g_velArrow = velArrow;
  g_corArrow = corArrow;

  g_crosshairLine = buildCrosshair(g_scene);
  g_orbit         = initOrbit(canvas);

  state3D.initPos = { ...latLonToVec3(DEFAULT_LAT, DEFAULT_LON) };
  resetSim3D();
}

// Expose to controls3d.js (loaded after this file)
window.state3D     = state3D;
window.startSim3D  = startSim3D;
window.pauseSim3D  = pauseSim3D;
window.resetSim3D  = resetSim3D;
window.renderFrame3D = renderFrame3D;
window.getOrbit    = () => g_orbit;
window.getEarthGroup = () => g_earthGroup;
window.getEarthMesh  = () => g_earthMesh;
window.getCamera     = () => g_camera;

window.addEventListener('load', initAll3D);
```
